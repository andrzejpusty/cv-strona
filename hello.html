Cel: Masz narzędzia, żeby pracować i łączyć się z serwerem.

sudo apt update && sudo apt upgrade -y sudo apt install -y git curl wget ca-certificates gnupg lsb-release unzip jq openssh-client
git config --global user.name "Andrzej Pusty" git config --global user.email "TWOJ_EMAIL"
wget -qO- https://packages.microsoft.com/keys/microsoft.asc | gpg --dearmor > packages.microsoft.gpg sudo install -o root -g root -m 644 packages.microsoft.gpg /etc/apt/trusted.gpg.d/ sudo sh -c 'echo "deb [arch=amd64,arm64,armhf] https://packages.microsoft.com/repos/code stable main" > /etc/apt/sources.list.d/vscode.list' sudo apt update sudo apt install -y code
ssh-keygen -t ed25519 -C "andrzej-vps" cat ~/.ssh/id_ed25519.pub
