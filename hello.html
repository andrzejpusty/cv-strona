ETAP 2 - Zabezpieczasz serwer i zakładasz usera deploy

Cel: Nie pracujesz na roocie. Logowanie tylko kluczem.

adduser deploy usermod -aG sudo deploy
mkdir -p /home/deploy/.ssh nano /home/deploy/.ssh/authorized_keys
chown -R deploy:deploy /home/deploy/.ssh chmod 700 /home/deploy/.ssh chmod 600 /home/deploy/.ssh/authorized_keys
ssh deploy@TWOJE_IP
sudo nano /etc/ssh/sshd_config PermitRootLogin no PasswordAuthentication no PubkeyAuthentication yes sudo systemctl restart ssh

ETAP 3 - Firewall + fail2ban

Cel: Podstawowe bezpieczeństwo "jak w firmie".

sudo apt -y install ufw sudo ufw default deny incoming sudo ufw default allow outgoing sudo ufw allow OpenSSH sudo ufw allow 80/tcp sudo ufw allow 443/tcp sudo ufw enable sudo ufw status
sudo apt -y install fail2ban sudo systemctl enable --now fail2ban sudo systemctl status fail2ban --no-pager

ETAP 4 - Docker + docker compose + struktura /opt

Cel: Serwer jest gotowy pod aplikacje w kontenerach.

sudo apt -y install ca-certificates curl gnupg sudo install -m 0755 -d /etc/apt/keyrings curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg sudo chmod a+r /etc/apt/keyrings/docker.gpg echo "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu $(. /etc/os-release && echo "$VERSION_CODENAME") stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null sudo apt update sudo apt -y install docker-ce docker-ce-cli containerd.io docker-buildx-plugin docker-compose-plugin
sudo usermod -aG docker deploy exit ssh deploy@TWOJE_IP
docker version docker compose version
sudo mkdir -p /opt/app /opt/infra sudo chown -R deploy:deploy /opt/app /opt/infra
